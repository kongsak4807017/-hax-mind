def ensure_dir(path: Path):

def read_env_file():
    env_path = ROOT / ".env"
    if not env_path.exists():
        return
    for line in env_path.read_text(encoding="utf-8", errors="ignore").splitlines():
        line = line.strip()
        if not line or line.startswith("#") or "=" not in line:

from dataclasses import dataclass
from typing import Optional

@dataclass
class Command:
    action: str
    target: Optional[str] = None
    args: Optional[str] = None

def parse_command(text: str) -> Command:
    text = text.strip()
    if not text.startswith("/"):
        return Command(action="message", args=text)
    parts = text.split(maxsplit=2)
    action = parts[0][1:]
    target = parts[1] if len(parts) > 1 else None
    args = parts[2] if len(parts) > 2 else None
    return Command(action=action, target=target, args=args)
'@

import json

from datetime import datetime

def log_event(event_type: str, content: str, topic: str = "general", importance: str = "normal"):
    path = RAW_LOG_DIR / f"{datetime.now().date()}.log"
    item = {
        "timestamp": datetime.now().isoformat(),
        "type": event_type,
        "topic": topic,
        "importance": importance,
        "content": content
    }
    with path.open("a", encoding="utf-8") as f:
        f.write(json.dumps(item, ensure_ascii=False) + "\n")

def summarize_today() -> str:
    src = RAW_LOG_DIR / f"{datetime.now().date()}.log"
    dst = SUMMARY_DIR / f"{datetime.now().date()}.md"
    if not src.exists():
        text = f"# Daily Summary: {datetime.now().date()}\n\nNo events found.\n"
        dst.write_text(text, encoding="utf-8")
        return text

    items = []
    for line in src.read_text(encoding="utf-8", errors="ignore").splitlines():
        try:
            items.append(json.loads(line))
        except Exception:
            continue

    lines = [f"# Daily Summary: {datetime.now().date()}", "", "## Highlights"]
    for item in items[:30]:
        lines.append(f"- [{item.get('topic','general')}] {item.get('content','')[:200]}")
    text = "\n".join(lines) + "\n"
    dst.write_text(text, encoding="utf-8")
    return text
'@

import subprocess
import sys

def run_tests(test_targets: list[str]) -> dict:
    results = []
    all_passed = True
    for target in test_targets:
        cmd = [sys.executable, "-m", "pytest", target]
        proc = subprocess.run(cmd, capture_output=True, text=True)
        passed = proc.returncode == 0
        all_passed = all_passed and passed
        results.append({
            "target": target,
            "passed": passed,
            "stdout": proc.stdout[-4000:],
            "stderr": proc.stderr[-4000:]
        })
    return {"all_passed": all_passed, "results": results}
'@

import json
import shutil
from pathlib import Path
from datetime import datetime

ROOT = Path(__file__).resolve().parent.parent
PROPOSALS_DIR = ROOT / "runtime" / "proposals"
APPLIED_DIR = ROOT / "runtime" / "applied"
FAILED_DIR = ROOT / "runtime" / "failed"
BACKUP_DIR = ROOT / "runtime" / "backups"

for d in [APPLIED_DIR, FAILED_DIR, BACKUP_DIR]:
    d.mkdir(parents=True, exist_ok=True)

def backup_files(files_to_modify: list[str], proposal_id: str) -> Path:
    backup_root = BACKUP_DIR / proposal_id
    backup_root.mkdir(parents=True, exist_ok=True)
    for file_path in files_to_modify:
        src = ROOT / file_path
        if src.exists() and src.is_file():
            dst = backup_root / file_path
            dst.parent.mkdir(parents=True, exist_ok=True)
            shutil.copy2(src, dst)
    return backup_root

def apply_placeholder_change(proposal: dict) -> dict:
    return {
        "applied": True,
        "summary": f"Placeholder apply completed for {proposal['title']}"
    }

def mark_status(proposal: dict, new_status: str):
    proposal["status"] = new_status
    path = PROPOSALS_DIR / f"{proposal['id']}.json"
    path.write_text(json.dumps(proposal, indent=2, ensure_ascii=False), encoding="utf-8")

def apply_proposal(proposal: dict) -> dict:
    backup_root = backup_files(proposal["files_to_modify"], proposal["id"])
    result = apply_placeholder_change(proposal)
    if result["applied"]:
        mark_status(proposal, "applied")
        (APPLIED_DIR / f"{proposal['id']}.txt").write_text(
            f"{datetime.now().isoformat()}\n{result['summary']}\nBackup: {backup_root}",
            encoding="utf-8"
        )
    else:
        mark_status(proposal, "failed")
        (FAILED_DIR / f"{proposal['id']}.txt").write_text(
            f"{datetime.now().isoformat()}\nApply failed\nBackup: {backup_root}",
            encoding="utf-8"
        )
    return result
'@

import shutil
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent
BACKUP_DIR = ROOT / "runtime" / "backups"

def rollback_proposal(proposal_id: str) -> dict:
    backup_root = BACKUP_DIR / proposal_id
    if not backup_root.exists():
        return {"ok": False, "message": "Backup not found"}

    restored = 0
    for file in backup_root.rglob("*"):
        if file.is_file():
            rel = file.relative_to(backup_root)
            dst = ROOT / rel
            dst.parent.mkdir(parents=True, exist_ok=True)
            shutil.copy2(file, dst)
            restored += 1

    return {"ok": True, "message": f"Restored {restored} files"}
'@

def generate_morning_report() -> str:
    proposals = list(PROPOSALS_DIR.glob("*.json"))
    applied = list(APPLIED_DIR.glob("*.txt"))
    failed = list(FAILED_DIR.glob("*.txt"))

    lines = [
        f"Morning Report — {datetime.now().isoformat()}",
        "",
        f"Pending proposals: {len(proposals)}",
        f"Applied changes: {len(applied)}",
        f"Failed changes: {len(failed)}",
        "",
        "Recommended next action:",
        "- Review pending proposals and approve the highest-impact medium-risk item."
    ]
    text = "\n".join(lines)
    (REPORTS_DIR / "morning_report.txt").write_text(text, encoding="utf-8")
    return text

import json
import os
from pathlib import Path

from telegram import Update
from telegram.ext import ApplicationBuilder, CommandHandler, ContextTypes

from engine.utils import read_env_file
from engine.self_improvement_engine import generate_improvement_proposal
from engine.morning_report import generate_morning_report
from engine.rollback_engine import rollback_proposal
from engine.memory_analyzer import log_event

ROOT = Path(__file__).resolve().parent.parent
PROPOSALS_DIR = ROOT / "runtime" / "proposals"

read_env_file()
TOKEN = os.environ.get("TELEGRAM_BOT_TOKEN")

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "HAX-Mind ready.\nCommands: /status /improve /proposals /approve <id> /reject <id> /rollback <id> /morning"
    )

async def status(update: Update, context: ContextTypes.DEFAULT_TYPE):
    count = len(list(PROPOSALS_DIR.glob("*.json")))
    await update.message.reply_text(f"System OK\nPending proposals: {count}")

async def improve(update: Update, context: ContextTypes.DEFAULT_TYPE):
    log_event("command", "/improve", topic="self_improvement", importance="high")
    proposal = generate_improvement_proposal()
    msg = (
        f"Proposal created\n"
        f"ID: {proposal['id']}\n"
        f"Title: {proposal['title']}\n"
        f"Risk: {proposal['risk']}\n"
        f"Problem: {proposal['problem']}\n"
        f"Solution: {proposal['solution']}"
    )
    await update.message.reply_text(msg)

async def proposals(update: Update, context: ContextTypes.DEFAULT_TYPE):
    files = list(PROPOSALS_DIR.glob("*.json"))
    if not files:
        await update.message.reply_text("No proposals.")
        return

    lines = []
    for file in files[-10:]:
        p = json.loads(file.read_text(encoding="utf-8"))
        lines.append(f"{p['id']} | {p['risk']} | {p['title']} | {p['status']}")
    await update.message.reply_text("\n".join(lines))

async def approve(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not context.args:
        await update.message.reply_text("Usage: /approve <proposal_id>")
        return
    pid = context.args[0]
    path = PROPOSALS_DIR / f"{pid}.json"
    if not path.exists():
        await update.message.reply_text("Proposal not found")
        return
    p = json.loads(path.read_text(encoding="utf-8"))
    p["status"] = "approved"
    path.write_text(json.dumps(p, indent=2, ensure_ascii=False), encoding="utf-8")
    log_event("approval", pid, topic="self_improvement", importance="high")
    await update.message.reply_text(f"Approved: {pid}")

async def reject(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not context.args:
        await update.message.reply_text("Usage: /reject <proposal_id>")
        return
    pid = context.args[0]
    path = PROPOSALS_DIR / f"{pid}.json"
    if not path.exists():
        await update.message.reply_text("Proposal not found")
        return
    p = json.loads(path.read_text(encoding="utf-8"))
    p["status"] = "rejected"
    path.write_text(json.dumps(p, indent=2, ensure_ascii=False), encoding="utf-8")
    await update.message.reply_text(f"Rejected: {pid}")

async def morning(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = generate_morning_report()
    await update.message.reply_text(text)

async def rollback(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not context.args:
        await update.message.reply_text("Usage: /rollback <proposal_id>")
        return
    result = rollback_proposal(context.args[0])
    await update.message.reply_text(result["message"])

def main():
    if not TOKEN:
        raise RuntimeError("TELEGRAM_BOT_TOKEN not found. Put it in .env")

    app = ApplicationBuilder().token(TOKEN).build()
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("status", status))
    app.add_handler(CommandHandler("improve", improve))
    app.add_handler(CommandHandler("proposals", proposals))
    app.add_handler(CommandHandler("approve", approve))
    app.add_handler(CommandHandler("reject", reject))
    app.add_handler(CommandHandler("rollback", rollback))
    app.add_handler(CommandHandler("morning", morning))
    app.run_polling()

if __name__ == "__main__":
    main()
